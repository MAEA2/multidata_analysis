alpha=np.arange(10E-10,2.0,0.01)

lin_E=[]
ridge_E=[]
lasso_E=[]

for a in alpha:
    linear=lm.LinearRegression()
    ridge = lm.Ridge(alpha=a)
    lasso = lm.Lasso(alpha=a)

    linear.fit(X_tr,y_tr)
    lasso.fit(X_tr,y_tr)
    ridge.fit(X_tr,y_tr)

    lin_error =0
    ridge_error=0
    lasso_error =0
    n_test=len(y_test)
    for i in range(n_test):
        lin_error += (y_test[i]-sum(X_test[i,]*linear.coef_)-linear.intercept_)**2/n_test
        ridge_error+=(y_test[i]-sum(X_test[i,]*ridge.coef_)-ridge.intercept_)**2/n_test
        lasso_error +=(y_test[i]-sum(X_test[i,]*lasso.coef_)-lasso.intercept_)**2/n_test
    lin_E.append(lin_error)
    ridge_E.append(ridge_error)
    lasso_E.append(lasso_error)

plt.hold(True);
plt.xlabel("alpha")
plt.ylabel("Mean Squared Error")
plt.plot(alpha,lin_E,label="linear") 
plt.plot(alpha,ridge_E,label="ridge")
plt.plot(alpha,lasso_E,label="lasso")
plt.legend(bbox_to_anchor=(1.05, 1), loc='upper left', borderaxespad=0)
